# 데이터 로드
digits_data = datasets.load_digits()
n = len(digits_data.images)

# 각 이미지는 8 * 8 배열로 표현된다.
# PCA에 대한 입력으로 사용하기 위해 이 배열을 평평하게 만든다.
image_data = digits_data.images.reshape((n, -1))

# PCA 변환기를 데이터셋에 피팅

pca_transformer = PCA(n_components=0.8)
